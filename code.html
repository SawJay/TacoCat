                        function getValues() {

                            let originalString = document.getElementById('inputString').value;
                            // reverse the string
                            let revString = reverseString(originalString)
                            // check original to see if equals reverse
                            let isPalindrome = checkForPalindrome(originalString, revString);
                            // display the 3 values in a message
                            displayResults(isPalindrome, originalString, revString);
                          
                          }
                          
                          function reverseString (originalString) {
                          
                            let reverseString = '';
                            
                            // radar
                            for (let index = originalString.length - 1; index >= 0; index--) {
                              
                              // r + a + d + a + r
                              reverseString += originalString[index];
                          
                            }
                          
                            return reverseString;
                          }
                          
                          function checkForPalindrome(originalString, revString) {
                          
                            if (originalString == revString) {
                              return true; 
                            } else {
                              return false;
                            }
                          
                          }
                          
                          function displayResults(isPalindrome) {
                          
                            if (isPalindrome == true ) {
                            document.getElementById('msg').innerHTML = `${isPalindrome} your word is a palindrome!`
                           }
                          
                           else {
                              document.getElementById('msg').innerHTML = `${isPalindrome} your word is not a palindrome.` 
                           }
                          
                            document.getElementById('alert').classList.remove('invisible'); 
                          }

The Code is strctured in four functions.

getValues()

The getValues function gathers user input from the HTML element with the id "originalString". If the input is empty, it prompts the user to enter text. After validation, it reverses the input using a decrementing loop, stores it in an array.

reverseString()

This function serves as a step for reversing the user's original word. It collects and processes the necessary data for the reversal.

checkForPalindrome()

The checkForPalindrome function compares originalString and revString values. It initializes a variable and determines if they are identical, assigning true if they match, and false otherwise. Subsequently, it returns this variable to the getValues function.

displayResults()

The displayResults function receives the isPalindrome variable. It then uses an if/else statement to determine if the checked value is true or false.
